class ExplodingProjectile:
    """Výbušná střela hráče — pomalejší, ale při zásahu poškodí
    všechny nepřátele v okolí 80px. Vizuálně modrá."""
    EXPLODE_RADIUS = 80
    DAMAGE = 2  # Poškodí nepřítele o 2 HP
 
    def __init__(self, x, y, target_x, target_y):
        self.x = x
        self.y = y
        speed = 9
        dx = target_x - x
        dy = target_y - y
        distance = math.hypot(dx, dy)
        self.dx = (dx / distance) * speed
        self.dy = (dy / distance) * speed
        self.rect = pygame.Rect(x - 10, y - 10, 20, 20)
        self.exploded = False
        self.explode_timer = 0   # vizuální efekt výbuchu
 
    def move(self):
        self.x += self.dx
        self.y += self.dy
        self.rect.center = (int(self.x), int(self.y))
 
    def draw(self):
        if self.explode_timer > 0:
            # Výbuch — kruh mizící pryč
            radius = int((1 - self.explode_timer / 15) * self.EXPLODE_RADIUS)
            alpha_surf = pygame.Surface((radius * 2 + 4, radius * 2 + 4), pygame.SRCALPHA)
            alpha = max(0, int(self.explode_timer / 15 * 200))
            pygame.draw.circle(alpha_surf, (0, 100, 255, alpha), (radius + 2, radius + 2), radius, 4)
            SCREEN.blit(alpha_surf, (int(self.x) - radius - 2, int(self.y) - radius - 2))
        else:
            pygame.draw.circle(SCREEN, DARK_BLUE, (int(self.x), int(self.y)), 10)
            pygame.draw.circle(SCREEN, (100, 150, 255), (int(self.x), int(self.y)), 6)
 
    def explode(self, enemies_list):
        """Poškodí nepřátele v dosahu výbuchu. Vrátí seznam zabitých."""
        self.exploded = True
        self.explode_timer = 15
        killed = []
        for e in enemies_list[:]:
            dist = math.hypot(e.x - self.x, e.y - self.y)
            if dist <= self.EXPLODE_RADIUS:
                e.hp -= self.DAMAGE
                if e.hp <= 0:
                    killed.append(e)
        return killed
 
    def collides_with(self, entity):
        return self.rect.colliderect(entity.rect)
